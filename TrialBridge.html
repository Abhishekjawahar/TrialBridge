// ZIP → approximate coords (US ZIP centroid lookup via free API)

// ═══════════════════════════════════════════════

async function zipToCoords(zip) {

  try {

    const r = await fetch(`https://api.zippopotam.us/us/${zip}`);

    if (!r.ok) return null;

    const d = await r.json();

    return { lat: parseFloat(d.places[0].latitude), lng: parseFloat(d.places[0].longitude), city: d.places[0]['place name'], state: d.places[0]['state abbreviation'] };

  } catch { return null; }

}


function haversineKm(lat1, lng1, lat2, lng2) {

  const R = 6371;

  const dLat = (lat2 - lat1) * Math.PI / 180;

  const dLng = (lng2 - lng1) * Math.PI / 180;

  const a = Math.sin(dLat/2)**2 + Math.cos(lat1*Math.PI/180) * Math.cos(lat2*Math.PI/180) * Math.sin(dLng/2)**2;

  return R * 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));

}

function kmToMiles(km) { return (km * 0.621371).toFixed(0); }

async function fetchTrials(condition, age, sex, phase) {

  const params = new URLSearchParams({

    'query.cond': condition,

    'filter.overallStatus': 'RECRUITING,ACTIVE_NOT_RECRUITING,ENROLLING_BY_INVITATION',

    'filter.advanced': `AREA[MinimumAge]RANGE[MIN, ${age} years] AND AREA[MaximumAge]RANGE[${age} years, MAX]`,

    'fields': 'NCTId,BriefTitle,BriefSummary,DetailedDescription,Phase,LeadSponsorName,StartDate,CompletionDate,Condition,EligibilityCriteria,HealthyVolunteers,Gender,MinimumAge,MaximumAge,LocationFacility,LocationCity,LocationState,LocationZip,LocationCountry,LocationGeoPoint,OverallStatus,StudyType,InterventionName,InterventionType',

    'pageSize': '30',

    'countTotal': 'true',

    'sort': 'LastUpdatePostDate:desc'

  });

  if (sex && sex !== '') params.set('filter.advanced', params.get('filter.advanced') + ` AND AREA[Gender](${sex} OR ALL)`);

  if (phase) params.set('filter.advanced', params.get('filter.advanced') + ` AND AREA[Phase]${phase}`);


  const r = await fetch(`${CT_BASE}?${params}`);

  if (!r.ok) throw new Error(`ClinicalTrials API error: ${r.status}`);

  const d = await r.json();

  return { studies: d.studies || [], total: d.totalCount || 0 };

}

// SCORING & ENRICHMENT

// ═══════════════════════════════════════════════

function scoreAndEnrichTrials(studies, userCoords, age) {

  return studies.map(study => {

    const p = study.protocolSection || {};

    const id = study.protocolSection?.identificationModule || {};

    const status = study.protocolSection?.statusModule || {};

    const desc = study.protocolSection?.descriptionModule || {};

    const elig = study.protocolSection?.eligibilityModule || {};

    const locs = study.protocolSection?.contactsLocationsModule?.locations || [];

    const design = study.protocolSection?.designModule || {};

    const arms = study.protocolSection?.armsInterventionsModule?.interventions || [];


    // Find closest location

    let closestDist = Infinity;

    let closestLoc = null;

    locs.forEach(loc => {

      if (loc.geoPoint && userCoords) {

        const d = haversineKm(userCoords.lat, userCoords.lng, loc.geoPoint.lat, loc.geoPoint.lon);

        if (d < closestDist) { closestDist = d; closestLoc = loc; }

      }

    });


    // Score 0–100

    let score = 50;

    if (closestDist < 80)  score += 25;  // < 50 miles

    else if (closestDist < 160) score += 15;

    else if (closestDist < 480) score += 5;

    const phase = (design.phases || [])[0] || '';

    if (phase === 'PHASE3') score += 15;

    if (phase === 'PHASE2') score += 8;

    const recStatus = status.overallStatus || '';

    if (recStatus === 'RECRUITING') score += 10;

    if (locs.length > 5) score += 5;

    score = Math.min(99, score);


    return {

      nctId: id.nctId || '',

      title: id.briefTitle || 'Untitled Study',

      summary: desc.briefSummary || '',

      detailedDesc: desc.detailedDescription || '',

      phase,

      status: recStatus,

      sponsor: p.sponsorCollaboratorsModule?.leadSponsor?.name || '',

      startDate: status.startDateStruct?.date || '',

      completionDate: status.completionDateStruct?.date || '',

      eligibility: elig.eligibilityCriteria || '',

      healthyVolunteers: elig.healthyVolunteers || '',

      gender: elig.sex || 'ALL',

      minAge: elig.minimumAge || '',

      maxAge: elig.maximumAge || '',

      locations: locs,

      closestLoc,

      closestDistMi: closestDist === Infinity ? null : parseFloat(kmToMiles(closestDist)),

      interventions: arms,

      studyType: design.studyType || '',

      score,

      aiSummary: null,

      aiEligibility: null,

      aiDoctorQs: null,

    };

  }).sort((a, b) => b.score - a.score);

}

async function startSearch() {

  const age       = parseInt(document.getElementById('age').value);

  const zip       = document.getElementById('zip').value.trim();

  const sex       = document.getElementById('sex')?.value || '';

  errEl.innerHTML = '';

  searchQuery = { condition, age, zip, sex, phase };

    setLoadingStep(3, 'active');

    const enriched = scoreAndEnrichTrials(studies, zipCoords, age);

    allTrials = enriched;

    filteredTrials = [...allTrials];

    setLoadingStep(3, 'done', `${enriched.length} matched`);


    // Step 4 — AI summaries for top 5 results

    setLoadingStep(4, 'active');

    const topFive = enriched.slice(0, 5);

    await Promise.all(topFive.map(t => enrichTrialWithAI(t)));

    setLoadingStep(4, 'done', `${topFive.length} explained`);


    // Show results

    showScreen('results');

    showScreen('intake');

    document.getElementById('intake-error').innerHTML = `<div class="error-card">Search failed: ${err.message}. Please check your connection and try again.</div>`;

}

function filterResults(type, btn) {

  if (type === 'all')        filteredTrials = [...allTrials];

  else if (type === 'phase3') filteredTrials = allTrials.filter(t => t.phase === 'PHASE3');

  else if (type === 'nearby') filteredTrials = allTrials.filter(t => t.closestDistMi !== null && t.closestDistMi <= 50);

  else if (type === 'high-match') filteredTrials = allTrials.filter(t => t.score >= 75);

  renderResults(searchQuery.zip);

}

function parseEligibilityAI(raw) {

  if (!raw) return { include: [], exclude: [] };

  const includeMatch = raw.match(/WHO CAN JOIN[:\s]*([\s\S]*?)(?=WHO CANNOT JOIN|$)/i);

  const excludeMatch = raw.match(/WHO CANNOT JOIN[:\s]*([\s\S]*?)$/i);

  const parseBullets = str => (str || '').split('\n').map(l => l.replace(/^[•\-\*\d\.]+\s*/, '').trim()).filter(Boolean);

  return {

    include: parseBullets(includeMatch?.[1] || ''),

    exclude: parseBullets(excludeMatch?.[1] || '')

  };

}


function parseDoctorQs(raw) {

  if (!raw) return [];

  return raw.split('\n').map(l => l.replace(/^\d+[\.\)]\s*/, '').trim()).filter(Boolean).slice(0, 4);

}
